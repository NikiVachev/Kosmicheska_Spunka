#include <iostream>
#include <thread>
#include <chrono>
#include <vector>


struct Coordinates {
    double x, y, z;
};

class PowerSignature {
public:
    std::string id;
    double strength;
    Coordinates coords;
};


class PowerLocator {
public:
    // returns a list of known signatures.
    std::vector<PowerSignature> scanArea() {
        // returns simulated data
        return {
            { "PS-ALPHA", 12.4, { 123.45, -67.89, 10.0 } },
            { "PS-BETA", 9.1,  { 321.00, -11.11,  5.0 } }
        };
    }
};


class PortalController {
public:
    bool initialize() {
        // initialize hardware 
        return true;
    }

    bool openPortalAt(const Coordinates& c) {
        // log and pretend success
        std::cout << "[portal] opening at (" << c.x << ", " << c.y << ", " << c.z << ")\\n";
        return true;
    }

    void closePortal() {
        std::cout << "[portal] closing\\n";
    }
};

int mainLoop(int loopSeconds) {
    PowerLocator locator;
    PortalController portal;

    if (!portal.initialize()) {
        std::cout << "Initialization failed (fictional).\\n";
        return -1;
    }

    while (true) {
        // Scan for power signatures
        auto found = locator.scanArea();

        // Select the strongest 
        if (!found.empty()) {
            PowerSignature chosen = found.front();
            for (auto &p : found) {
                if (p.strength > chosen.strength) chosen = p;
            }

            // Open portal to the chosen 
            if (portal.openPortalAt(chosen.coords)) {
                std::cout << "[main] portal opened to " << chosen.id << "\\n";
                // Hold portal for a short interval
                std::this_thread::sleep_for(std::chrono::seconds(loopSeconds));
                portal.closePortal();
            }
        } else {
            std::cout << "[main] no signatures found; retrying...\\n";
        }

        // Wait until the next scheduled attempt 
        std::this_thread::sleep_for(std::chrono::seconds(loopSeconds));
    }

    return 0;
}
